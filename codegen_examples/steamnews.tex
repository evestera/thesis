#[derive(Default, Debug, Clone, PartialEq, Serialize, Deserialize)]
struct SteamAppNews {
    appnews: Appnews,
}

#[derive(Default, Debug, Clone, PartialEq, Serialize, Deserialize)]
struct Appnews {
    appid: i64,
    newsitems: Vec<Newsitem>,
    count: i64,
}

#[derive(Default, Debug, Clone, PartialEq, Serialize, Deserialize)]
struct Newsitem {
    gid: String,
    title: String,
    url: String,
    is_external_url: bool,
    author: String,
    contents: String,
    feedlabel: String,
    date: i64,
    feedname: String,
    feed_type: i64,
    appid: i64,
}
\end{minted}
